97  status = WAIT_FOR_MOTORS_TO_FINNISH; // set the state machine to wait for the motors to finish
98  nextStatus = Running; // the next status to move to when the motors finish
99  break;
100  case Running:
101  Serial.println(" Running State Machine "+String(millis()));
102  // Set up a non-blocking 1000 ms delay
103  status=WAIT_FOR_TIME;
104  nextTime=millis()+1000;
105  // After 1000 ms, come back to this state
106  nextStatus=Running;
107  break;
108  case WAIT_FOR_TIME:
109  // Check to see if enough time has elapsed
110  if(nextTime<=millis()){
111  // if the time is up, move on to the next state
112  status = nextStatus;
113  }
114  break;
115  case WAIT_FOR_MOTORS_TO_FINNISH:
116  if (motor1->isInterpolationDone() && motor2->isInterpolationDone()
117  && motor3->isInterpolationDone()) {
118  status = nextStatus;
119  }
120  break;
121  case Halting:
122  // save state and enter safe mode
123  Serial.println("Halting State machine");
124  digitalWrite(EMITTER_PIN, 0);
